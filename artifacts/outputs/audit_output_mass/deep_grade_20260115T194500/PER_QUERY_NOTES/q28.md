## Minimal Fix

**File:** `backend/src/services/core/languageEnforcement.service.ts`

Add post-generation sanitization:

```typescript
function sanitizeAnswer(answer: string, targetLanguage: string): string {
  if (targetLanguage !== 'en') return answer;

  // Detect and translate/quote Portuguese phrases
  const ptPhrasePattern = /[""]([^""]+)[""]|'([^']+)'/g;

  return answer.replace(ptPhrasePattern, (match, quoted) => {
    if (isProbablyPortuguese(quoted)) {
      const translation = translatePhrase(quoted, 'pt', 'en');
      return `"${quoted}" (${translation})`;
    }
    return match;
  });
}

function isProbablyPortuguese(text: string): boolean {
  const ptIndicators = ['ção', 'ência', 'ção', 'loja', 'não', 'são'];
  return ptIndicators.some(ind => text.toLowerCase().includes(ind));
}
```
